        function parseImport() {

            const text = document.getElementById('import-area').value;

            if(!text.trim()) return;


            const lines = text.split('\n');

            let found = 0;

            const dateRegex = /(\d{2}[\/-]\d{2})/g;

            const codeRegex = /\b([A-Z]{3})\b/g;


            lines.forEach(line => {

                const dates = line.match(dateRegex);

                const codes = line.match(codeRegex);


                if(dates && codes && codes.length >= 2) {

                    let esc = null;

                    for(let c of codes) {

                        if(AIRPORTS[c]) { esc = c; break; }

                    }

                    if(!esc && !['CDG', 'ORY', 'AFR', 'ARR', 'DEP'].includes(codes[1])) {

                        esc = codes[1];

                    }


                    if(esc) {

                        const info = AIRPORTS[esc] || { pays: esc, taux: 170, zone: 'LC' };

                        const on = 1;

                        let val = (info.zone === 'MC') ? Math.max(0, (on - 0.5) * info.taux) : on * info.taux;

                        appData.rotations.push({

                            date: dates[0],

                            code: esc,

                            pays: info.pays,

                            on: on,

                            taux: info.taux,

                            zone: info.zone || 'LC',

                            total: val

                        });

                        found++;

                    }

                }

            });


            if(found > 0) {

                renderTable();

                calculate();

                document.getElementById('import-area').value = "";

                alert(found + " vols détectés !");
